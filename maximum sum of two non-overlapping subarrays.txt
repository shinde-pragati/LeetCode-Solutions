class Solution {
public:
    int maxSumTwoNoOverlap(vector<int>& nums, int firstLen, int secondLen) {
        int n = nums.size();
        vector<int> prefixSum(n + 1, 0);
        
        // Build prefix sums
        for (int i = 0; i < n; i++) {
            prefixSum[i + 1] = prefixSum[i] + nums[i];
        }
        
        // Helper function to get sum of subarray from i to j (inclusive)
        auto getSum = [&](int start, int length) {
            return prefixSum[start + length] - prefixSum[start];
        };
        
        // Case 1: firstLen before secondLen
        int maxSum1 = 0;
        int maxFirstSum = 0;
        for (int i = 0; i + firstLen + secondLen <= n; i++) {
            // Update the best sum for first subarray ending at or before i
            maxFirstSum = max(maxFirstSum, getSum(i, firstLen));
            // Second subarray starts at i + firstLen
            int secondSum = getSum(i + firstLen, secondLen);
            maxSum1 = max(maxSum1, maxFirstSum + secondSum);
        }
        
        // Case 2: secondLen before firstLen
        int maxSum2 = 0;
        int maxSecondSum = 0;
        for (int i = 0; i + secondLen + firstLen <= n; i++) {
            // Update the best sum for second subarray ending at or before i
            maxSecondSum = max(maxSecondSum, getSum(i, secondLen));
            // First subarray starts at i + secondLen
            int firstSum = getSum(i + secondLen, firstLen);
            maxSum2 = max(maxSum2, maxSecondSum + firstSum);
        }
        
        return max(maxSum1, maxSum2);
    }
};
