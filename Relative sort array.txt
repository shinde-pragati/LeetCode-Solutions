#include <vector>
#include <map>
#include <algorithm>
using namespace std;

class Solution {
public:
    vector<int> relativeSortArray(vector<int>& arr1, vector<int>& arr2) {
        map<int,int> freq;
        
        // Count frequency of arr1
        for(int x : arr1)
            freq[x]++;
        
        vector<int> result;
        
        // Add elements in order of arr2
        for(int x : arr2){
            while(freq[x] > 0){
                result.push_back(x);
                freq[x]--;
            }
        }
        
        // Add remaining elements in sorted order
        for(auto &p : freq){
            while(p.second > 0){
                result.push_back(p.first);
                p.second--;
            }
        }
        
        return result;
    }
};
