class Solution {
public:
    vector<string> uncommonFromSentences(string s1, string s2) {
        unordered_map<string, int> count;
        vector<string> result;
        
        auto countWords = [&](string& s) {
            istringstream iss(s);
            string word;
            while (iss >> word) {
                count[word]++;
            }
        };

        countWords(s1);
        countWords(s2);
   
        for (auto& pair : count) {
            if (pair.second == 1) {
                result.push_back(pair.first);
            }
        }
        
        return result;
    }
};
