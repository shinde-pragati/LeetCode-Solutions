class Solution {
public:
    int mirrorReflection(int p, int q) {
     
        
        int g = gcd(p, q);
        int k = q / g;  
        int m = p / g;  
       
        if (m % 2 == 0) {
            return 2;
        }
        if (k % 2 == 0) {
            return 0;
        }
        return 1;
    }
    
private:
    int gcd(int a, int b) {
        while (b) {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }
};
