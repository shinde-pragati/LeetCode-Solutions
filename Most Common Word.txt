class Solution {
public:
    string mostCommonWord(string paragraph, vector<string>& banned) {
        // Convert to lowercase
        for (char& c : paragraph) {
            c = tolower(c);
        }
        
        // Create a set of banned words for O(1) lookup
        unordered_set<string> bannedSet(banned.begin(), banned.end());
        
        // Count frequencies of valid words
        unordered_map<string, int> freq;
        string word = "";
        int maxFreq = 0;
        string result = "";
        
        // Process each character
        for (int i = 0; i <= paragraph.length(); i++) {
            // If current character is a letter, add to current word
            if (i < paragraph.length() && isalpha(paragraph[i])) {
                word += paragraph[i];
            } 
            // If we hit a non-letter or end of string, process the word
            else if (!word.empty()) {
                // Only count if not banned
                if (bannedSet.find(word) == bannedSet.end()) {
                    freq[word]++;
                    if (freq[word] > maxFreq) {
                        maxFreq = freq[word];
                        result = word;
                    }
                }
                word = ""; // Reset for next word
            }
        }
        
        return result;
    }
};
