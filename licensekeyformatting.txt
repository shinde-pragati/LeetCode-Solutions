class Solution {
public:
    string licenseKeyFormatting(string s, int k) {
       string result;

       for (char c : s) {
        if (c != '-') {
            result += toupper(c);
        }
       } 
       int len = result.length();
       for (int i = len - k; i > 0; i -= k) {
        result.insert(i, 1, '-');
       }
       return result;
    }
};
