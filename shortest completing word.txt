class Solution {
public:
    string shortestCompletingWord(string licensePlate, vector<string>& words) {
        vector<int> licenseCount(26, 0);
        for (char c : licensePlate) {
            if (isalpha(c)) {
                licenseCount[tolower(c) - 'a']++;
            }
        }
        
        string result = "";
        int minLength = INT_MAX;
        
        for (const string& word : words) {
            if (isCompletingWord(word, licenseCount)) {
                if (word.length() < minLength) {
                    minLength = word.length();
                    result = word;
                }
            }
        }
        
        return result;
    }
    
private:
    bool isCompletingWord(const string& word, const vector<int>& licenseCount) {
        vector<int> wordCount(26, 0);
        
        for (char c : word) {
            wordCount[tolower(c) - 'a']++;
        }
        
        for (int i = 0; i < 26; i++) {
            if (wordCount[i] < licenseCount[i]) {
                return false;
            }
        }
        
        return true;
    }
};
