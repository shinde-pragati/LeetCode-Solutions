#include <vector>
#include <climits>
using namespace std;

class Solution {
public:
    int maxAbsValExpr(vector<int>& arr1, vector<int>& arr2) {
        int n = arr1.size();
        int ans = 0;

        vector<int> sign = {1,-1};

        for(int s1 : sign){
            for(int s2 : sign){

                int mx = INT_MIN;
                int mn = INT_MAX;

                for(int i = 0; i < n; i++){
                    int val = arr1[i]*s1 + arr2[i]*s2 + i;

                    mx = max(mx,val);
                    mn = min(mn,val);
                }

                ans = max(ans, mx - mn);
            }
        }

        return ans;
    }
};
