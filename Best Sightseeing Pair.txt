class Solution {
public:
    int maxScoreSightseeingPair(vector<int>& values) {
        int maxScore = 0;
        int best_i_value = values[0] + 0;
        
        for (int j = 1; j < values.size(); j++) {
         
            maxScore = max(maxScore, best_i_value + values[j] - j);
         
            best_i_value = max(best_i_value, values[j] + j);
        }
        
        return maxScore;
    }
};
