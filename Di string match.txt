class Solution {
public:
    vector<int> diStringMatch(string s) {
        int n = s.length();
        vector<int> result(n + 1);
        
        int low = 0;
        int high = n;
        
        for (int i = 0; i < n; i++) {
            if (s[i] == 'I') {
                result[i] = low++;
            } else { // 'D'
                result[i] = high--;
            }
        }
      
        result[n] = low; 
        
        return result;
    }
};
