class Solution {
public:
    int numUniqueEmails(vector<string>& emails) {
        unordered_set<string> uniqueEmails;
        
        for (string& email : emails) {
            string cleanEmail;
            bool ignore = false;
            bool domain = false;
            
            for (char c : email) {
                if (c == '@') {
                    domain = true;
                    cleanEmail += c;
                } 
                else if (domain) {
                    cleanEmail += c;
                } 
                else { 
                    if (c == '+') {
                        ignore = true;
                    } 
                    else if (c != '.' && !ignore) {
                        cleanEmail += c;
                    }
                }
            }
            
            uniqueEmails.insert(cleanEmail);
        }
        
        return uniqueEmails.size();
    }
};
