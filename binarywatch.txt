class Solution {
public:
    vector<string> readBinaryWatch(int turnedOn) {
        vector<string> result;
        
  
        for (int h = 0; h <= min(4, turnedOn); h++) {
            int m = turnedOn - h;
            
  
            if (m < 0 || m > 6) continue;
            
  
            vector<int> hours = generateNumbers(4, h, 0, 11);
            
  
            vector<int> minutes = generateNumbers(6, m, 0, 59);
            
  
            for (int hour : hours) {
                for (int minute : minutes) {
                    result.push_back(formatTime(hour, minute));
                }
            }
        }
        
        return result;
    }
    
private:
      vector<int> generateNumbers(int totalBits, int bits, int minVal, int maxVal) {
        vector<int> result;
       
        int maxNum = (1 << totalBits) - 1;
        
        for (int num = 0; num <= maxNum; num++) {
            if (__builtin_popcount(num) == bits && num >= minVal && num <= maxVal) {
                result.push_back(num);
            }
        }
        
        return result;
    }
    
    string formatTime(int hour, int minute) {
        string h = to_string(hour);
        string m = (minute < 10) ? "0" + to_string(minute) : to_string(minute);
        return h + ":" + m;
    }
};
