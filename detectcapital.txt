class Solution {
public:
    bool detectCapitalUse(string word) {
        int n = word.length();
        int capitals = 0;
        
        for (char c : word) {
            if (isupper(c)) {
                capitals++;
            }
        }

        if (capitals == n) return true;
     
        if (capitals == 0) return true;
       
        if (capitals == 1 && isupper(word[0])) return true;
        
        return false;
    }
};
